   static char[]  str=null;
   public static void main(String s[]) {
      if(s.length==0)
	System.exit(-1);

       int end=str.length-1;

       System.out.print("Original string=");

	  begin++;
	  end--;       
       }

       System.out.print("\n" + "Reversed string=");

    }
