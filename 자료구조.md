  ```c
  //head 삭제
  if empty, do nothing
  temp = head
  head = head.next

  ```

  ```c
  // tail 삭제
  if empty, do nothing
  Vertex pre = head
  temp = head.next
  while (temp.next != null)
    pre = pre.next
    tmp = pre.next
  pre.next = null
  delete temp, tail = pre
  ```

  ```c
  //특정 인덱스 삭제   i = 인덱스
  ```
